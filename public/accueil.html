let configPlats = []; async function chargerMenu() { const res = await fetch('plats.json'); const data = await res.json(); configPlats = data.plats; // Génère les boutons de la grille principale const grid = document.querySelector('.grid-menus'); grid.innerHTML = configPlats.map(p => ` `).join(''); } function sel(id) { const plat = configPlats.find(p => p.id === id); // Initialise les options selon la config du plat opt = { p: plat.id, c: (plat.hasCuisson ? 'Apoint' : 'N/A'), a: (plat.hasAcc ? 'Crudités' : 'N/A'), t: (plat.hasTaille ? 'Adulte' : 'N/A') }; document.getElementById('titre-sel').innerText = plat.nom; document.getElementById('options').style.display = 'block'; // Affiche/Masque les blocs selon le JSON document.getElementById('bloc-cuisson').style.display = plat.hasCuisson ? 'block' : 'none'; document.getElementById('bloc-acc').style.display = plat.hasAcc ? 'block' : 'none'; document.getElementById('bloc-taille').style.display = plat.hasTaille ? 'block' : 'none'; updateUI(); } window.addEventListener('DOMContentLoaded', chargerMenu);
